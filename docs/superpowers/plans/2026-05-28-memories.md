# Memories Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Persist user / team / org context across conversations with explicit user + admin controls, GDPR cascade, PII gating, provenance, and decay. Memories integrate into chat (extract on turn complete, recall before turn), assistants (assistant-scope), task workers (repo-scope), and RAG (`memory.search` tool).

**Architecture:** Extend existing module `server/api/src/ai_portal/memory/` (already has `model.py`, `router.py`, `workers/`). Add pluggable sub-packages: `memory/extractors/`, `memory/recallers/`, `memory/stores/`, `memory/policies/`, plus `memory/service.py`, `memory/repository.py`, `memory/schemas.py`, `memory/decay.py`, `memory/jobs.py`. All cross-cutting concerns (auth, audit, usage, webhooks, RBAC, blob storage, register_deleter, register_exporter) come from `control_plane`. LLM extraction + embeddings come from `gateway` facade. Sensitive-category + PII gating reuses gateway `guardrails` providers.

**Tech Stack:** Python 3.12, FastAPI, SQLAlchemy 2.x async, Alembic, Pydantic v2, asyncpg, pgvector (default vector store), qdrant-client (optional), rank-bm25 (hybrid recaller), pytest + pytest-asyncio, respx.

**Spec:** `docs/superpowers/specs/2026-05-28-memories-design.md`
**Depends on:**
- `2026-05-28-control-plane.md` (hard — RBAC, audit, usage, webhooks, settings, BlobStore, register_deleter, register_exporter)
- `2026-05-28-gateway.md` (hard — `complete()` for extraction, `embed()` for vectors, guardrail providers for PII gate)
- `2026-05-28-rag.md` (soft — only when `rag_backed` store provider is chosen)

# repository.py
from sqlalchemy import select, and_, or_, update
from ai_portal.memory.model import Memory, MemoryScope, MemoryUse, ScopeKind

class MemoryRepo:
    def __init__(self, session): self.s = session

    async def add(self, m: Memory) -> Memory:
        self.s.add(m); await self.s.flush(); return m

    async def get(self, mid: str) -> Memory | None:
        return (await self.s.execute(select(Memory).where(Memory.id == mid))).scalar_one_or_none()

    async def soft_delete(self, mid: str):
        from datetime import datetime
        await self.s.execute(update(Memory).where(Memory.id == mid).values(deleted_at=datetime.utcnow()))
        await self.s.flush()

    async def list_for_actor(self, org_id, actor_user_id, team_ids, assistant_id=None,
                              conversation_id=None, type=None, q=None, limit=200):
        clauses = [Memory.org_id == org_id, Memory.deleted_at.is_(None)]
        scope_or = [
            and_(Memory.scope_kind == ScopeKind.user,
                 Memory.scope_ids_json.contains([actor_user_id])),
            and_(Memory.scope_kind == ScopeKind.org, Memory.scope_ids_json.contains([org_id])),
        ]
        if team_ids:
            scope_or.append(and_(Memory.scope_kind == ScopeKind.team,
                                 or_(*[Memory.scope_ids_json.contains([t]) for t in team_ids])))
        if assistant_id:
            scope_or.append(and_(Memory.scope_kind == ScopeKind.assistant,
                                 Memory.scope_ids_json.contains([assistant_id])))
        if conversation_id:
            scope_or.append(and_(Memory.scope_kind == ScopeKind.conversation,
                                 Memory.scope_ids_json.contains([conversation_id])))
        clauses.append(or_(*scope_or))
        if type: clauses.append(Memory.type == type)
        if q: clauses.append(Memory.text.ilike(f"%{q}%"))
        rows = await self.s.execute(select(Memory).where(and_(*clauses)).limit(limit))
        return list(rows.scalars())

    async def vector_search(self, org_id, embedding, scope_filter, limit=20):
        clauses = [Memory.org_id == org_id, Memory.deleted_at.is_(None)]
        # cosine distance via pgvector
        rows = await self.s.execute(
            select(Memory, Memory.embedding.cosine_distance(embedding).label("dist"))
            .where(and_(*clauses))
            .order_by("dist").limit(limit)
        )
        return [(m, float(d)) for m, d in rows.all()]

    async def record_use(self, memory_id, query_hash, response_message_id, score):
        self.s.add(MemoryUse(memory_id=memory_id, query_text_hash=query_hash,
                             response_message_id=response_message_id, score=score))
        await self.s.flush()

    async def bulk_delete(self, ids: list[str]):
        from datetime import datetime
        await self.s.execute(update(Memory).where(Memory.id.in_(ids))
                             .values(deleted_at=datetime.utcnow()))
        await self.s.flush()
```

from dataclasses import dataclass, field

@dataclass
class Turn:
    role: str
    content: str
    turn_id: str
    ts: float

@dataclass
class ExtractScope:
    org_id: str
    actor_user_id: str
    scope_kind: str            # user|conversation|assistant|team|org
    scope_id: str
    conversation_id: str | None
    assistant_id: str | None = None

@dataclass
class ExtractOpts:
    model: str = "claude-sonnet-4-6"
    allowed_types: list[str] = field(default_factory=lambda: ["fact","preference","entity","episode","procedure","relation"])
    block_sensitive_categories: list[str] = field(default_factory=list)
    confidence_floor: float = 0.4
    dedupe: bool = True

@dataclass
class Candidate:
    type: str
    text: str
    confidence: float
    source_turn_ids: list[str]
    tags: list[str] = field(default_factory=list)

@runtime_checkable
class Extractor(Protocol):
    name: str
    async def extract(self, turns: list[Turn], scope: ExtractScope, opts: ExtractOpts) -> list[Candidate]: ...
```

# registry.py
from .protocol import Extractor

_REG: dict[str, Extractor] = {}

def register(e: Extractor) -> None:
    _REG[e.name] = e

def get(name: str) -> Extractor:
    if name not in _REG:
        raise KeyError(f"unknown extractor: {name}")
    return _REG[name]

def list_names() -> list[str]:
    return sorted(_REG)

# llm_default.py
import json
from ai_portal.gateway import complete as gw_complete
from ai_portal.memory.extractors.protocol import Extractor, Turn, ExtractScope, ExtractOpts, Candidate
from ai_portal.memory.extractors.registry import register

SYS = """Extract memories from conversation. Output JSON only.
- One memory per durable fact, preference, entity, episode, procedure.
- Skip greetings, small talk, ephemeral context.
- Skip sensitive categories listed in BLOCK.
- type in {fact, preference, entity, relation, episode, procedure}
- confidence in [0,1]
- text: third person, atomic, <240 chars.
Return: {"memories": [{type, text, confidence, source_turn_ids: [...]}]}"""

class LlmDefaultExtractor:
    name = "llm_default"

    async def extract(self, turns: list[Turn], scope: ExtractScope, opts: ExtractOpts) -> list[Candidate]:
        block = ",".join(opts.block_sensitive_categories) or "none"
        user = "BLOCK: " + block + "\n\nTURNS:\n" + "\n".join(
            f"[{t.turn_id}] {t.role}: {t.content}" for t in turns
        )
        res = await gw_complete({
            "model": opts.model,
            "messages": [
                {"role": "system", "content": [{"type": "text", "text": SYS}]},
                {"role": "user", "content": [{"type": "text", "text": user}]},
            ],
            "response_format": {"type": "json_schema"},
            "temperature": 0.0,
            "max_tokens": 1024,
            "metadata": {"module": "memory", "phase": "extract"},
        })
        text = "".join(b["text"] for b in res.content if b.get("type") == "text")
        try:
            parsed = json.loads(text)
        except json.JSONDecodeError:
            return []
        out = []
        for item in parsed.get("memories", []):
            if item.get("type") not in opts.allowed_types: continue
            if item.get("confidence", 0.0) < opts.confidence_floor: continue
            out.append(Candidate(
                type=item["type"],
                text=item["text"][:240],
                confidence=float(item["confidence"]),
                source_turn_ids=item.get("source_turn_ids", []),
                tags=item.get("tags", []),
            ))
        return out

```python
# vector_pgvector.py
import hashlib, math, time
from ai_portal.gateway import embed as gw_embed
from ai_portal.memory.recallers.protocol import Recaller, RecallOpts, Recalled
from ai_portal.memory.recallers.registry import register
from ai_portal.memory.repository import MemoryRepo

def _recency(last_used_at, now) -> float:
    if not last_used_at: return 0.0
    age_d = max(0, (now - last_used_at.timestamp()) / 86400.0)
    return math.exp(-age_d / 30.0)   # 30d half-life-ish

class VectorPgvectorRecaller:
    name = "vector_pgvector"

    def __init__(self, session):
        self.repo = MemoryRepo(session)

    async def recall(self, query: str, scope, opts: RecallOpts) -> list[Recalled]:
        emb = (await gw_embed([query], model="text-embedding-3-small")).data[0]
        rows = await self.repo.vector_search(scope.org_id, emb, scope_filter=scope, limit=opts.top_k * 3)
        now = time.time()
        out: list[Recalled] = []
        for m, dist in rows:
            vec_score = 1.0 - dist
            rec = _recency(m.last_used_at or m.created_at, now)
            score = (1 - opts.recency_weight - opts.importance_weight) * vec_score \
                  + opts.recency_weight * rec \
                  + opts.importance_weight * m.importance
            out.append(Recalled(memory_id=m.id, text=m.text, score=score,
                                explain={"vector": vec_score, "recency": rec,
                                         "importance": m.importance, "why": "vector cosine"}))
        out.sort(key=lambda r: r.score, reverse=True)
        return out[:opts.top_k]

# gdpr.py
from sqlalchemy import select, delete
from ai_portal.control_plane import register_deleter, register_exporter, emit_audit
from ai_portal.memory.model import Memory, MemoryUse

async def delete_for_actor(session, org_id: str, actor) -> int:
    if actor.kind != "user": return 0
    rows = (await session.execute(
        select(Memory.id).where(Memory.org_id == org_id,
                                Memory.actor_owner_json["id"].astext == actor.id)
    )).scalars().all()
    if not rows: return 0
    await session.execute(delete(MemoryUse).where(MemoryUse.memory_id.in_(rows)))
    await session.execute(delete(Memory).where(Memory.id.in_(rows)))
    await session.flush()
    await emit_audit("memory.gdpr.deleted", resource_type="user",
                     resource_id=actor.id, payload={"count": len(rows)})
    return len(rows)

async def export_for_actor(session, org_id: str, actor) -> dict:
    from sqlalchemy import select
    rows = (await session.execute(
        select(Memory).where(Memory.org_id == org_id,
                             Memory.actor_owner_json["id"].astext == actor.id)
    )).scalars().all()
    return {"memories": [
        {"id": m.id, "type": m.type.value, "text": m.text,
         "scope_kind": m.scope_kind.value, "scope_ids": m.scope_ids_json,
         "source_conversation_id": m.source_conversation_id,
         "source_turn_ids": m.source_turn_ids_json,
         "extractor_model": m.extractor_model,
         "created_at": m.created_at.isoformat()}
        for m in rows
    ]}

register_deleter("memory", delete_for_actor)
register_exporter("memory", export_for_actor)
```
